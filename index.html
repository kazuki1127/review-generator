// URLパラメータを取得

const params = new URLSearchParams(window.location.search);

const purpose = params.get("purpose");

const liked = params.get("liked");

const improvement = params.get("improvement");


// テンプレートにデータを埋め込む

const templates = {

  "カット＆シャンプー": "「{purpose}で来店しました。{liked} またお願いしたいです。」",

  "フロントカット": "「前髪を整えてもらいたくて{purpose}で伺いました。{liked} {improvement}」",

  // 他のテンプレートも追加

};


// 生成文

let reviewText = templates[purpose] || "口コミを作成できませんでした。";

reviewText = reviewText.replace("{purpose}", purpose).replace("{liked}", liked).replace("{improvement}", improvement || "");


// 表示

const resultDiv = document.getElementById("js-result");

const generatedReview = document.createElement("div");

generatedReview.id = "generated-review";

generatedReview.style.cssText = "border: 1px solid #ddd; padding: 10px; margin: 20px 0;";

generatedReview.innerText = reviewText;

resultDiv.appendChild(generatedReview);


// コピーボタンを作成

const copyButton = document.createElement("button");

copyButton.innerText = "口コミをコピーする";

copyButton.onclick = copyReview;

resultDiv.appendChild(copyButton);


// コピー機能

function copyReview() {

  navigator.clipboard.writeText(reviewText).then(() => {

    const alertMessage = document.createElement("p");

    alertMessage.innerText = "口コミをコピーしました！";

    alertMessage.style.color = "green";

    resultDiv.appendChild(alertMessage);

    setTimeout(() => alertMessage.remove(), 3000);

  });

}
