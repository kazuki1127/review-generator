<body>

  <h1>口コミ生成ツール</h1>

  <p>以下の内容で口コミを生成しました。ボタンを押してコピーしてください。</p>

  <div id="generated-review" style="border: 1px solid #ddd; padding: 10px; margin: 20px 0;"></div>

  <button onclick="copyReview()">口コミをコピーする</button>


  <script>

    // URLパラメータを取得

    const params = new URLSearchParams(window.location.search);

    const purpose = params.get("purpose");

    const liked = params.get("liked");

    const improvement = params.get("improvement");


    // テンプレートにデータを埋め込む

    const templates = {

      "カット＆シャンプー": "「{purpose}で来店しました。{liked} またお願いしたいです。」",

      "フロントカット": "「前髪を整えてもらいたくて{purpose}で伺いました。{liked} {improvement}」",

      // 他のテンプレートも追加

    };


    // 生成文

    let reviewText = templates[purpose] || "口コミを作成できませんでした。";

    reviewText = reviewText.replace("{purpose}", purpose).replace("{liked}", liked).replace("{improvement}", improvement || "");


    // 表示

    document.getElementById("generated-review").innerText = reviewText;


    // コピー機能

    function copyReview() {

      navigator.clipboard.writeText(reviewText).then(() => {

        alert("口コミをコピーしました！");

      });

    }

  </script>

</body>
